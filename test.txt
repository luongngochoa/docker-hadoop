ENV TOMCAT_VERSION 9.0.56

apt-get install -y curl

# Install Tomcat for serving Tez UI
RUN curl -LO https://downloads.apache.org/tomcat/tomcat-9/v9.0.56/bin/apache-tomcat-9.0.56.tar.gz && \
    tar -xvf apache-tomcat-9.0.56.tar.gz && \
    mv apache-tomcat-9.0.56 /opt/tomcat && \
    rm apache-tomcat-9.0.56.tar.gz

# Download and deploy Tez UI
RUN curl -LO https://repo1.maven.org/maven2/org/apache/tez/tez-ui/0.10.1/tez-ui-0.10.1.war && \
    mv tez-ui-0.10.1.war /opt/tomcat/webapps/tez-ui.war

# Expose Tez UI port
EXPOSE 8080

# Start Tomcat and serve Tez UI
CMD ["/opt/tomcat/bin/catalina.sh", "run"]
